-- hospitals table (NEW — run this in Supabase SQL Editor)
CREATE TABLE hospitals (
  id            UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  name          TEXT NOT NULL,
  address       TEXT NOT NULL,
  phone_number  TEXT NOT NULL,       -- Twilio will call this number
  latitude      FLOAT NOT NULL,
  longitude     FLOAT NOT NULL,
  is_active     BOOLEAN DEFAULT true,
  created_at    TIMESTAMPTZ DEFAULT now()
);

-- Enable Row Level Security (public read for the agent)
ALTER TABLE hospitals ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Allow service role full access" ON hospitals
  FOR ALL USING (true);
